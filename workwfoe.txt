name: Sentinel DevSecOps CI

on:
  push:
    branches: [ main ]

jobs:
  build-and-scan:
    runs-on: ubuntu-latest

    steps:
    # 1️⃣ Checkout Repository
    - name: Checkout Code
      uses: actions/checkout@v4
      with:
        fetch-depth: 0

    # 2️⃣ Python SAST Scan (Bandit)
    - name: Bandit Security Scan
      run: |
        pip install bandit
        bandit -r ./app -ll

    # 3️⃣ Build Docker Images
    - name: Build Docker Stack
      run: |
        docker compose build

    # 4️⃣ Debug Step (shows images built in runner)
    - name: List Docker Images
      run: docker images

    # 5️⃣ Container Security Scan (Trivy)
    - name: Run Trivy Scan (Backend)
      uses: aquasecurity/trivy-action@master
      with:
        image-ref: 'sentinel-backend:latest'
        format: 'table'
        exit-code: '1'
        severity: 'CRITICAL'

    # 6️⃣ SonarCloud Code Quality Scan
    - name: SonarCloud Scan
      uses: SonarSource/sonarqube-scan-action@master
      env:
        GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}
        SONAR_TOKEN: ${{ secrets.SONAR_TOKEN }}
      # Login to Docker Hub
      # Login to Docker Hub
    - name: Login to Docker Hub
      uses: docker/login-action@v3
      with:
        username: ${{ secrets.DOCKER_USERNAME }}
        password: ${{ secrets.DOCKER_PASSWORD }}

    # Tag images for Docker Hub
    - name: Tag Docker Images
      run: |
        docker tag sentinel-backend:latest ${{ secrets.DOCKER_USERNAME }}/sentinel-backend:latest
        docker tag sentinel-frontend:latest ${{ secrets.DOCKER_USERNAME }}/sentinel-frontend:latest

    # Push images
    - name: Push to Docker Hub
      run: |
        docker push ${{ secrets.DOCKER_USERNAME }}/sentinel-backend:latest
        docker push ${{ secrets.DOCKER_USERNAME }}/sentinel-frontend:latest
